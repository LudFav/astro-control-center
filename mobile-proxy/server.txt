/*
  Astro Control Center - Mobile proxy for OnStepX
  Run on Android (Termux) or any local machine.
*/

const http = require("http");
const { URL } = require("url");

const PORT = Number(process.env.PORT || 8787);
const ONSTEP_BASE = (process.env.ONSTEP_BASE || "http://192.168.0.1").replace(/\/+$/, "");
const ALLOW_ORIGIN = process.env.ALLOW_ORIGIN || "*";
const REQUEST_TIMEOUT_MS = Number(process.env.REQUEST_TIMEOUT_MS || 4000);
const API_TOKEN = (process.env.API_TOKEN || "").trim();
const BIND_HOST = (process.env.BIND_HOST || "127.0.0.1").trim();

function sendJson(res, code, data) {
  res.writeHead(code, {
    "Content-Type": "application/json; charset=utf-8",
    "Access-Control-Allow-Origin": ALLOW_ORIGIN,
    "Access-Control-Allow-Methods": "GET,POST,OPTIONS",
    "Access-Control-Allow-Headers": "Content-Type, Authorization",
    "Access-Control-Allow-Private-Network": "true"
  });
  res.end(JSON.stringify(data));
}

async function fetchWithTimeout(url, options = {}, timeoutMs = REQUEST_TIMEOUT_MS) {
  const timeout = new Promise((_, reject) =>
    setTimeout(() => reject(new Error("timeout")), timeoutMs)
  );
  return Promise.race([fetch(url, options), timeout]);
}

function isAuthorized(req) {
  if (!API_TOKEN) return true;
  const header = req.headers.authorization || "";
  return header === `Bearer ${API_TOKEN}`;
}

async function tryGetStatus() {
  const paths = ["/api/v1/status", "/status", "/api/status"];
  for (const p of paths) {
    try {
      const r = await fetchWithTimeout(`${ONSTEP_BASE}${p}`, { method: "GET" });
      if (!r.ok) continue;
      const data = await r.json();
      return { ok: true, endpoint: p, data };
    } catch (_e) {
      // try next
    }
  }
  return { ok: false };
}

async function tryPostByCandidates(type, payloads) {
  const map = {
    goto: ["/api/v1/goto", "/goto", "/api/goto"],
    abort: ["/api/v1/abort", "/abort", "/api/abort"],
    slew: ["/api/v1/slew", "/slew", "/api/slew"]
  };
  const paths = map[type] || [];
  for (const p of paths) {
    for (const payload of payloads) {
      try {
        const r = await fetchWithTimeout(`${ONSTEP_BASE}${p}`, {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify(payload)
        });
        if (!r.ok) continue;
        return { ok: true, endpoint: p };
      } catch (_e) {
        // try next
      }
    }
  }
  return { ok: false };
}

function readBody(req) {
  return new Promise((resolve, reject) => {
    let raw = "";
    req.on("data", (chunk) => {
      raw += chunk;
      if (raw.length > 1_000_000) {
        reject(new Error("payload_too_large"));
      }
    });
    req.on("end", () => {
      if (!raw) return resolve({});
      try {
        resolve(JSON.parse(raw));
      } catch (_e) {
        reject(new Error("invalid_json"));
      }
    });
    req.on("error", reject);
  });
}

const server = http.createServer(async (req, res) => {
  if (req.method === "OPTIONS") {
    res.writeHead(204, {
      "Access-Control-Allow-Origin": ALLOW_ORIGIN,
      "Access-Control-Allow-Methods": "GET,POST,OPTIONS",
      "Access-Control-Allow-Headers": "Content-Type, Authorization",
      "Access-Control-Allow-Private-Network": "true"
    });
    return res.end();
  }

  const url = new URL(req.url, `http://${req.headers.host}`);

  if (req.method === "GET" && url.pathname === "/health") {
    return sendJson(res, 200, { ok: true, onstepBase: ONSTEP_BASE });
  }

  if (!isAuthorized(req)) {
    return sendJson(res, 401, { ok: false, error: "unauthorized" });
  }

  if (req.method === "GET" && (url.pathname === "/api/v1/status" || url.pathname === "/status" || url.pathname === "/api/status")) {
    const out = await tryGetStatus();
    if (!out.ok) return sendJson(res, 502, { ok: false, error: "onstep_unreachable" });
    return sendJson(res, 200, out.data);
  }

  if (req.method === "POST" && (url.pathname === "/api/v1/goto" || url.pathname === "/goto" || url.pathname === "/api/goto")) {
    try {
      const body = await readBody(req);
      const payloads = [
        body,
        { ra: body.ra, dec: body.dec, name: body.name, epoch: body.epoch || "J2000" },
        { raHours: body.ra, decDeg: body.dec, target: body.name, frame: body.epoch || "J2000" }
      ];
      const out = await tryPostByCandidates("goto", payloads);
      if (!out.ok) return sendJson(res, 502, { ok: false, error: "goto_failed" });
      return sendJson(res, 200, { ok: true, endpoint: out.endpoint });
    } catch (e) {
      return sendJson(res, 400, { ok: false, error: e.message || "bad_request" });
    }
  }

  if (req.method === "POST" && (url.pathname === "/api/v1/abort" || url.pathname === "/abort" || url.pathname === "/api/abort")) {
    const out = await tryPostByCandidates("abort", [{}, { action: "abort" }, { cmd: "abort" }]);
    if (!out.ok) return sendJson(res, 502, { ok: false, error: "abort_failed" });
    return sendJson(res, 200, { ok: true, endpoint: out.endpoint });
  }

  if (req.method === "POST" && (url.pathname === "/api/v1/slew" || url.pathname === "/slew" || url.pathname === "/api/slew")) {
    try {
      const body = await readBody(req);
      const payloads = [
        body,
        { direction: body.direction || body.dir, rate: "guide", action: "pulse", durationMs: body.durationMs || 400 },
        { dir: body.direction || body.dir, speed: "guide", ms: body.durationMs || 400 }
      ];
      const out = await tryPostByCandidates("slew", payloads);
      if (!out.ok) return sendJson(res, 502, { ok: false, error: "slew_failed" });
      return sendJson(res, 200, { ok: true, endpoint: out.endpoint });
    } catch (e) {
      return sendJson(res, 400, { ok: false, error: e.message || "bad_request" });
    }
  }

  return sendJson(res, 404, { ok: false, error: "not_found" });
});

server.listen(PORT, BIND_HOST, () => {
  console.log(`Astro mobile proxy listening on ${BIND_HOST}:${PORT}`);
  console.log(`Forward target ONSTEP_BASE=${ONSTEP_BASE}`);
  console.log(`Auth token ${API_TOKEN ? "enabled" : "disabled"}`);
});
